üìÑ example-webhook.sh 
# Generate timestamp and signature
TIMESTAMP=$(date +%s)
PAYLOAD='{"type":"venue_update","venues":[{"name":"Green Cafe","type":"restaurant","address":{"street":"123 Main St","city":"Zurich","postal_code":"8001","country":"CH"}}]}'
SIGNATURE=$(echo -n "${TIMESTAMP}.${PAYLOAD}" | openssl dgst -sha256 -hmac "your_webhook_secret" | cut -d' ' -f2)

# Send the webhook request
curl -X POST https://europe-west6-planted-availability.cloudfunctions.net/partnerWebhook \
  -H "Authorization: Bearer pad_live_your_api_key" \
  -H "Content-Type: application/json" \
  -H "X-Planted-Timestamp: ${TIMESTAMP}" \
  -H "X-Planted-Signature: ${SIGNATURE}" \
  -d "${PAYLOAD}"

# Response
# {
#   "message": "Data received and staged for processing",
#   "batch_id": "batch_abc123xyz",
#   "status": "pending_review",
#   "stats": { "venues": 1, "dishes": 0, "promotions": 0, "total": 1 }
# }
